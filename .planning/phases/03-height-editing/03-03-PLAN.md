```csharp
/// <summary>
/// Processes height editing input when viewport is hovered.
/// Per D-01: Real-time editing mode.
/// Per D-02: Left mouse down to edit, release to end stroke.
/// Per D-03: Per-frame update of affected pixels.
/// </summary>
public void UpdateEditing(float deltaTime)
{
    if (!IsViewportHovered || IsViewportInteracting)
        return;

    if (terrainManager?.CurrentTerrain == null)
        return;

    var io = ImGui.GetIO();
    var editorState = EditorState.Instance;
    var heightEditor = HeightEditor.Instance;

    // Get current hit point from brush preview logic
    var camera = GetActiveCamera();
    if (camera == null)
        return;

    NumericsVector2 mousePos = io.MousePos;
    float toolbarHeight = GetToolbarHeight();
    float infoHeight = GetInfoBarHeight();
    float viewY = ContentRect.Y + toolbarHeight;
    float viewHeight = Math.Max(0.0f, ContentRect.Height - toolbarHeight - infoHeight);
    NumericsVector2 viewPos = new NumericsVector2(ContentRect.X, viewY);
    NumericsVector2 viewSize = new NumericsVector2(ContentRect.Width, viewHeight);

    var (rayOrigin, rayDirection) = TerrainRaycast.ScreenToWorldRay(
        mousePos.X, mousePos.Y,
        viewPos.X, viewPos.Y,
        viewSize.X, viewSize.Y,
        camera);

    var hitPoint = TerrainRaycast.RayTerrainIntersection(rayOrigin, rayDirection, terrainManager);

    // Check for edit start (D-02: left mouse down)
    if (io.MouseDown[0] && !isEditing && hitPoint != null)
    {
        isEditing = true;
        editHitPoint = hitPoint;
        heightEditor.BeginStroke(editorState.CurrentTool.ToString(), hitPoint.Value, terrainManager);
    }

    // Apply edit during drag (D-01, D-03)
    if (isEditing && io.MouseDown[0] && hitPoint != null)
    {
        heightEditor.ApplyStroke(hitPoint.Value, terrainManager, deltaTime);
        editHitPoint = hitPoint;
    }

    // End edit on release (D-02)
    if (!io.MouseDown[0] && isEditing)
    {
        isEditing = false;
        editHitPoint = null;
        heightEditor.EndStroke();
    }
}
```
