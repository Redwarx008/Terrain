    ```csharp
    public void ApplyStroke(Vector3 worldPosition, TerrainManager terrainManager, float frameTime)
    {
        if (!isStrokeActive || currentTool == null)
            return;

        // Get the entity containing this position
        var entity = terrainManager.GetEntityAtPosition(worldPosition.X, worldPosition.Z);
        if (entity == null)
            return;

        // Get height data cache for this entity
        var heightData = entity.HeightDataCache;
        if (heightData == null)
            return;

        // Convert world position to chunk-local pixel coordinates
        int pixelX = (int)MathF.Round(worldPosition.X - entity.WorldOffset.X);
        int pixelZ = (int)MathF.Round(worldPosition.Z - entity.WorldOffset.Z);

        // Get brush parameters
        var brushParams = BrushParameters.Instance;
        float brushRadius = brushParams.Size * 0.5f;
        float brushInnerRadius = brushRadius * brushParams.EffectiveFalloff;

        // Build edit context for this specific chunk
        var context = new HeightEditContext
        {
            HeightData = heightData,
            DataWidth = entity.HeightmapWidth,
            DataHeight = entity.HeightmapHeight,
            CenterX = pixelX,
            CenterZ = pixelZ,
            BrushRadius = brushRadius,
            BrushInnerRadius = brushInnerRadius,
            Strength = brushParams.Strength,
            FrameTime = frameTime
        };

        // Apply the tool
        currentTool.Apply(ref context);

        // Sync to GPU and propagate to adjacent chunks if needed
        // Per CONTEXT.md: Cross-chunk editing syncs overlap regions
        terrainManager.UpdateHeightData((int)worldPosition.X, (int)worldPosition.Z, brushRadius);
    }
    ```

    The key difference is:
    - Get the specific entity at the world position
    - Convert world coordinates to chunk-local coordinates
    - Apply editing to that entity's height cache
