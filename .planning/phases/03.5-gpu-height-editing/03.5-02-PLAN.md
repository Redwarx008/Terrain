/// <summary>
/// Creates GPU texture from height data with UAV support for compute shader writes.
/// Uses R16_Float format for UAV compatibility (R16_UNorm does not support UAV on all GPUs).
/// </summary>
public static Texture CreateHeightmapTextureUAV(
    GraphicsDevice graphicsDevice,
    ushort[] heightData,
    int width,
    int height)
{
    if (graphicsDevice == null)
        throw new ArgumentNullException(nameof(graphicsDevice));

    if (heightData == null || heightData.Length == 0)
        throw new ArgumentNullException(nameof(heightData));

    // R16_Float supports UAV; R16_UNorm may not on all hardware
    // Convert ushort to half-float for upload
    var halfData = new Half[width * height];
    for (int i = 0; i < heightData.Length; i++)
    {
        halfData[i] = new Half(heightData[i] / (float)ushort.MaxValue);
    }

    var texture = Texture.New2D(
        graphicsDevice,
        width,
        height,
        PixelFormat.R16_Float,
        TextureFlags.ShaderResource | TextureFlags.UnorderedAccess);

    texture.SetData(graphicsDevice.Context.CommandList, new ReadOnlySpan<Half>(halfData));

    Log.Info($"Created UAV heightmap texture: {width}x{height}");
    return texture;
}
```

#nullable enable

using System;
using Stride.Core.Mathematics;
using Stride.Graphics;
using Stride.Rendering;
using Terrain.Editor.Rendering;

namespace Terrain.Editor.Services;

/// <summary>
/// GPU-based height editor using compute shaders for parallel brush operations.
/// Replaces CPU-based HeightEditor for better performance with large brushes.
/// </summary>
public sealed class GpuHeightEditor : IDisposable
{
    private static readonly Lazy<GpuHeightEditor> _instance = new(() => new());

    public static GpuHeightEditor Instance => _instance.Value;

    private HeightEditComputeDispatcher? dispatcher;
    private RenderContext? renderContext;

    private int currentEditMode;
    private float flattenTargetHeight;
    private bool isStrokeActive;

    /// <summary>
    /// Initializes the GPU editor with a render context. Must be called before any editing.
    /// </summary>
    public void Initialize(RenderContext context)
    {
        renderContext = context;
        dispatcher = new HeightEditComputeDispatcher();
        dispatcher.Initialize(context);
    }

    /// <summary>
    /// Begins a new editing stroke.
    /// </summary>
    /// <param name="toolName">Tool name: "Raise", "Lower", "Smooth", or "Flatten"</param>
    /// <param name="worldPosition">World position where stroke begins</param>
    /// <param name="terrainManager">Terrain manager for height queries</param>
    public void BeginStroke(string toolName, Vector3 worldPosition, TerrainManager terrainManager)
    {
        if (dispatcher == null)
            throw new InvalidOperationException("GpuHeightEditor not initialized. Call Initialize() first.");

        isStrokeActive = true;

        currentEditMode = toolName switch
        {
            "Raise" => 0,
            "Lower" => 1,
            "Smooth" => 2,
            "Flatten" => 3,
            _ => throw new ArgumentException($"Unknown tool: {toolName}", nameof(toolName))
        };

        // For Flatten, sample target height at click position
        if (toolName == "Flatten")
        {
            var height = terrainManager.GetHeightAtPosition(worldPosition.X, worldPosition.Z);
            flattenTargetHeight = height ?? 0f;
        }
    }

    /// <summary>
    /// Applies the current stroke at the given position using GPU compute.
    /// </summary>
    public void ApplyStroke(
        RenderDrawContext drawContext,
        Vector3 worldPosition,
        TerrainManager terrainManager,
        float frameTime)
    {
        if (!isStrokeActive || dispatcher == null)
            return;

        var terrainEntity = terrainManager.TerrainEntities.Count > 0
            ? terrainManager.TerrainEntities[0]
            : null;

        if (terrainEntity == null)
            return;

        // Convert world position to pixel coordinates
        int pixelX = (int)MathF.Round(worldPosition.X);
        int pixelZ = (int)MathF.Round(worldPosition.Z);

        // Get brush parameters
        var brushParams = BrushParameters.Instance;
        float brushRadius = brushParams.Size * 0.5f;
        float brushInnerRadius = brushRadius * brushParams.EffectiveFalloff;

        // Find the slice containing this pixel
        var slice = terrainEntity.GetSliceForPixel(pixelX, pixelZ);
        if (slice == null)
            return;

        // Convert to slice-local coordinates
        int localX = pixelX - slice.StartSampleX;
        int localZ = pixelZ - slice.StartSampleZ;

        // Dispatch compute shader
        dispatcher.Dispatch(
            drawContext,
            slice.Texture,
            localX,
            localZ,
            brushRadius,
            brushInnerRadius,
            brushParams.Strength,
            frameTime,
            currentEditMode,
            flattenTargetHeight);

        // Mark slice as dirty for CPU cache sync
        slice.IsDirty = true;

        // Update terrain bounds
        terrainEntity.MarkHeightRegionDirty(pixelX, pixelZ, brushRadius);
    }

    /// <summary>
    /// Ends the current stroke.
    /// </summary>
    public void EndStroke()
    {
        isStrokeActive = false;
    }

    public void Dispose()
    {
        dispatcher?.Dispose();
        dispatcher = null;
    }
}
```

2. Requires Initialize() call with RenderContext before editing
3. Converts world coordinates to slice-local coordinates
4. Dispatches to the correct slice's UAV texture
5. Maintains IsDirty flag for CPU cache synchronization
