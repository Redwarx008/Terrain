   - `stage int EditMode` - 0=Raise, 1=Lower, 2=Smooth, 3=Flatten

5. Implement `Compute()` override:
   - Get texel coordinates from `streams.DispatchThreadId.xy`
   - Boundary check against texture dimensions
   - Distance check from brush center
   - Apply falloff
   - Switch on EditMode:

     - Lower: `newHeight = currentHeight - Strength * FrameTime * 0.015 * falloff`
     - Smooth: 3x3 kernel average, blend toward average
     - Flatten: blend toward TargetHeight
   - Clamp result to [0, 1] and write to HeightmapSlice

Thread count: 8x8 (64 threads per group), matching TerrainBuildLodMap.sdsl pattern.

Class structure:
```csharp
#nullable enable
using System;
using Stride.Core.Mathematics;
using Stride.Graphics;
using Stride.Rendering;

namespace Terrain.Editor.Services;

public sealed class HeightEditComputeDispatcher : IDisposable
{
    private const int ThreadCountX = 8;
    private const int ThreadCountY = 8;

    private ComputeEffectShader? heightEditEffect;

    public void Initialize(RenderContext renderContext)
    {
        heightEditEffect = new ComputeEffectShader(renderContext)
        {
            ShaderSourceName = "TerrainHeightEdit",
            ThreadNumbers = new Int3(ThreadCountX, ThreadCountY, 1),
        };
    }

    public void Dispatch(
        RenderDrawContext drawContext,
        Texture heightmapSlice,
        int brushCenterX, int brushCenterY,
        float brushRadius, float brushInnerRadius,
        float strength, float frameTime,
        int editMode, float targetHeight)
    {
        // Calculate thread groups
        int threadGroupCountX = (heightmapSlice.Width + ThreadCountX - 1) / ThreadCountX;
        int threadGroupCountY = (heightmapSlice.Height + ThreadCountY - 1) / ThreadCountY;

        var commandList = drawContext.CommandList;

        // Transition to UAV state
        commandList.ResourceBarrierTransition(heightmapSlice, GraphicsResourceState.UnorderedAccess);

        // Set parameters
        heightEditEffect.ThreadGroupCounts = new Int3(threadGroupCountX, threadGroupCountY, 1);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.HeightmapSlice, heightmapSlice);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.BrushCenter, new Int2(brushCenterX, brushCenterY));
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.BrushRadius, brushRadius);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.BrushInnerRadius, brushInnerRadius);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.Strength, strength);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.FrameTime, frameTime);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.EditMode, editMode);
        heightEditEffect.Parameters.Set(TerrainHeightEditKeys.TargetHeight, targetHeight);

        heightEditEffect.Draw(drawContext);

        // Transition back to SRV state
        commandList.ResourceBarrierTransition(heightmapSlice, GraphicsResourceState.PixelShaderResource);
    }

    public void Dispose()
    {
        heightEditEffect?.Dispose();
        heightEditEffect = null;
    }
}
```

Note: The `TerrainHeightEditKeys` class is auto-generated by Stride's shader compiler from the .sdsl file. The executor should verify the Keys are available after shader compilation.
