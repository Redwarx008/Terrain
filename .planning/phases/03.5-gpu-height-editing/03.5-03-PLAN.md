```csharp
/// <summary>
/// Syncs GPU height data back to CPU cache for undo/redo and file export.
/// Call this after GPU editing operations complete.
/// </summary>
public void SyncFromGpu(CommandList commandList)
{
    if (heightDataCache == null || terrainEntities.Count == 0)
        return;

    var entity = terrainEntities[0];
    foreach (var slice in entity.Slices)
    {
        if (!slice.IsDirty)
            continue;

        // Read back from GPU to CPU
        var texture = slice.Texture;
        var pixelCount = slice.Width * slice.Height;

        // Download from GPU - R16_Float format
        var data = new Half[pixelCount];
        texture.GetData(commandList, data);

        // Copy to height cache, converting Half back to ushort
        for (int row = 0; row < slice.Height; row++)
        {
            int srcOffset = row * slice.Width;
            int dstOffset = (slice.StartSampleZ + row) * heightDataWidth + slice.StartSampleX;

            for (int x = 0; x < slice.Width; x++)
            {
                float heightFloat = (float)data[srcOffset + x];
                heightDataCache[dstOffset + x] = (ushort)Math.Clamp(heightFloat * ushort.MaxValue, 0, ushort.MaxValue);
            }
        }

        slice.IsDirty = false;
    }
}
```

```csharp
/// <summary>
/// Processes height editing input when viewport is hovered.
/// Per D-01: Real-time editing mode.
/// Per D-02: Left mouse down to edit, release to end stroke.
/// Per D-03: Per-frame update of affected pixels.
/// </summary>
public void UpdateEditing(NumericsVector2 viewPos, NumericsVector2 viewSize)
{
    // D-02: Only block editing during camera interaction (right/middle button), not left button editing.
    if (!IsViewportHovered)
        return;

    // Still block if camera is being controlled (right/middle mouse held)
    var io = ImGui.GetIO();
    bool isCameraInteracting = io.MouseDown[(int)ImGuiMouseButton.Right] || io.MouseDown[(int)ImGuiMouseButton.Middle];
    if (isCameraInteracting)
        return;

    if (!terrainManager.HasTerrainLoaded)
        return;

    var editorState = EditorState.Instance;
    var gpuHeightEditor = GpuHeightEditor.Instance;

    // Use currentHitPoint from RenderBrushPreview
    if (currentHitPoint == null)
        return;

    // Need graphics context for GPU dispatch
    if (graphicsContext == null)
        return;

    // Check for edit start (D-02: left mouse down)
    if (io.MouseDown[0] && !isEditing)
    {
        isEditing = true;
        gpuHeightEditor.BeginStroke(editorState.CurrentTool.ToString(), currentHitPoint.Value, terrainManager);
    }

    // Apply edit during drag (D-01, D-03)
    if (isEditing && io.MouseDown[0])
    {
        float deltaTime = (float)io.DeltaTime;

        // Create RenderDrawContext for GPU dispatch
        var renderDrawContext = new RenderDrawContext(
            graphicsContext.RenderContext,
            graphicsContext.CommandList,
            graphicsContext.ResourceGroupAllocator);

        gpuHeightEditor.ApplyStroke(renderDrawContext, currentHitPoint.Value, terrainManager, deltaTime);
    }

    // End edit on release (D-02)
    if (!io.MouseDown[0] && isEditing)
    {
        isEditing = false;
        gpuHeightEditor.EndStroke();

        // Sync GPU changes back to CPU cache for undo/save
        terrainManager.SyncFromGpu(graphicsContext.CommandList);
    }
}
```
